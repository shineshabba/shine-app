      # АДАПТИРОВАТЬ маппинг под реальные поля из db.py:

  Предположительный шаблон — адаптировать под реальные имена функций из xui.py:

  ```python
  import asyncio
  from fastapi import APIRouter, Depends, HTTPException, Request
  from auth import validate_init_data
  from models import ConfigResponse, UserProfile
  from main import limiter
  # ЗАМЕНИТЬ на реальные имена функций из db.py и xui.py:
  from services.db import get_user_by_tg_id  # или аналог
  from services.xui import get_client_vless_url  # или аналог — читать xui.py

  router = APIRouter()

  @router.get("/config", response_model=ConfigResponse)
  @limiter.limit("60/minute")
  async def get_config(
      request: Request,
      user_data: dict = Depends(validate_init_data)
  ):
      """
      GET /api/config — VLESS-ссылка (API-02, D-06).
      403 если подписка не активна.
      """
      tg_user_id = user_data["id"]
      request.state.tg_user_id = tg_user_id

      # Проверить статус подписки
      loop = asyncio.get_event_loop()
      user = await loop.run_in_executor(None, get_user_by_tg_id, tg_user_id)

      if user is None:
          raise HTTPException(status_code=404, detail="User not found")

      is_active = bool(user.get("subscription_active") or user.get("is_active"))
      if not is_active:
          raise HTTPException(status_code=403, detail="Subscription not active")

      # Получить VLESS-конфиг из x-ui
      # АДАПТИРОВАТЬ вызов под реальную функцию из xui.py:
      vless_url = await loop.run_in_executor(None, get_client_vless_url, tg_user_id)

      if not vless_url:
          raise HTTPException(status_code=404, detail="VPN config not found")

      return ConfigResponse(vless_url=vless_url)
  ```

  from fastapi import FastAPI, Request
  from fastapi.middleware.cors import CORSMiddleware
  from slowapi import Limiter, _rate_limit_exceeded_handler
  from slowapi.errors import RateLimitExceeded
  from slowapi.middleware import SlowAPIMiddleware

  from auth import get_user_id_for_limiter
  from routers import health, me, config

  limiter = Limiter(key_func=get_user_id_for_limiter)

  app = FastAPI(
      title="Shine App API",
      version="1.0.0",
      description="Backend API for Shine VPN Telegram Mini App",
  )

  app.state.limiter = limiter
  app.add_exception_handler(RateLimitExceeded, _rate_limit_exceeded_handler)
  app.add_middleware(SlowAPIMiddleware)

  app.add_middleware(
      CORSMiddleware,
      allow_origins=["http://localhost:5173", "http://localhost:3000"],
      allow_origin_regex=r"https://.*\.vercel\.app",
      allow_credentials=True,
      allow_methods=["GET", "OPTIONS"],
      allow_headers=["X-Telegram-Init-Data", "Content-Type"],
  )

  app.include_router(health.router, prefix="/api")
  app.include_router(me.router, prefix="/api")
  app.include_router(config.router, prefix="/api")

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP Request → /api/me | Любой запрос с X-Telegram-Init-Data header; данные не доверенные |
| /api/me → services/db.py | Функция получает tg_user_id только из validate_init_data результата |
| /api/config → services/xui.py | Функция вызывается только после проверки subscription_active |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Spoofing | GET /api/me tg_user_id | mitigate | tg_user_id берётся из user_data["id"] где user_data = result of validate_init_data; исключает передачу чужого user_id |
| T-02-02 | Information Disclosure | GET /api/config для неактивного | mitigate | 403 возвращается ДО вызова xui.py; VLESS-ссылка никогда не генерируется для неактивных |
| T-02-03 | Denial of Service | GET /api/config тяжёлый запрос к x-ui | mitigate | slowapi 60/min per user на endpoint; x-ui вызов только после auth проверки |
| T-02-04 | Information Disclosure | stack trace в 500 ответе | mitigate | FastAPI по умолчанию не раскрывает traceback в JSON ответе; только {"detail":"Internal Server Error"} |
| T-02-05 | Tampering | SQLite concurrent write | accept | Backend только читает; WAL mode в db.py предотвращает locked errors; write через бот |
</threat_model>
